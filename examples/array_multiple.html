class A
{
	int i;
}

A a[4][4];
<<<a[2][3]>>>;
<<<2 => a[2][3].i>>>;

int i[4][4];
<<<54 => i[2][3]>>>;

float f[4][4];
<<<f[2][3]>>>;
<<<12.345 => f[2][3]>>>;

complex c[4][4];
<<<c[2][3]>>>;
<<<c[2][3].re>>>;

polar p[4][4];
<<<p[2][3]>>>;
<<<p[2][3].mod>>>;

Vec3 v[4][4];
<<<v[2][3]>>>;
<<<v[2][3].x>>>;

Vec4 w[4][4];
<<<w[2][3]>>>;
<<<w[2][3].w>>>;
