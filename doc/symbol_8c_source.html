63 void free_Symbols()
64 {
65  int i;
66  for(i = 0; i < SIZE; i++) {
67  S_Symbol s = hashtable[i];
68  if(s) {
69  S_Symbol tmp, next = s;

94 S_Symbol insert_symbol(c_constr name)
95 {
96  S_Symbol syms = NULL, sym;
97  int index;
98  if(!name)
99  return NULL;
100  index = hash(name) % SIZE;
101  syms = hashtable[index];
102  for(sym = syms; sym; sym = sym->next)
103  if (streq(sym->name, name))
104  return sym;
105  sym = mksymbol(name, syms);
106  hashtable[index] = sym;
107  return sym;
108 }
